闭包单体模式

			单体（singleton）模式是javascript中最基本但又最有用的模式之一，它可能比其他任何模式都常用。
			
			这种模式提供了一种将代码组织为一个逻辑单元的手段，这个逻辑单元中的代码可以通过单一的变量进行访问。
			
			通过确保单体对象只存在一份实例，你就可以确信自己的所有代码使用的都是同样的全局资源。
			
			单体（singleton）模式有： 1、简单单体		 2、闭包单体		3、惰性单体		4、分支单体
			

			
			/*
			 * 闭包单体模式：
			 * 借用闭包创建单体：注：闭包的主要目的是：保护数据的，不受外界影响【创建一个自己的私有的作用域(function(){块级作用域})()】！
			 */
			
			var BXH = {} //命名空间
			BXH.Singleton = (function(){ //块级作用域：格式：(function(){私有成员 return{外界能访问的}})()；立即执行函数赋给BXH.Singleton
				
				//1、添加私有的成员(属性 或 方法)，外界无法访问！【这就是闭包的好处】，因为安全！
				var a = 1;
				var b = "1";
				var c = a + b;
				var mu = true;
				var name = '小明'
				var DaTe = new Date();
				var F1 = function()
				{
					alert('我是私有的方法，只有在返回对象里面输出我，外面才能访问我！');
				}
				
				//2、把块级作用域里的执体结果返回【就是外界想要访问的数据，只能是下面(return返回的)有的，才能访问！】
				return{
					attr1 : a, 
					attr2 : name,
					dat : DaTe,
					m : mu,
					zdy : '我是在返回对象中自定义的属性，不是私有的！请随便用我！',
					
					method1 : function()
					{
						return F1();
					},
					
					method2 : function()
					{
						alert('向外界发出：C的值是' + c +'，但是：外界无法访问，上面的 a 和 b');
					}
				}; 
			})();
			
			BXH.Singleton.method1();
			
			BXH.Singleton.method2();
			
			alert(BXH.Singleton.attr1 + BXH.Singleton.attr2 + BXH.Singleton.dat + BXH.Singleton.zdy);
			
			alert('试试到低能不能访问：'+ BXH.Singleton.a + BXH.Singleton.b + BXH.Singleton.a1 + BXH.Singleton.name + BXH.Singleton.dates);
