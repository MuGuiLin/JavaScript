			//用原型：prototype 自己写一个实现测循环遍历多维数组的方法！
			
			//var Arr = [1,2,3,[4,[5,[6,[7,[8]]]]]] ; //六维数组
			var Arr = [12,'你好',['abc',123,false,new Date()],true,'张三',1024,['hello',123,['abc',456,false],'add',789],'还好',true,new Date()]; //三维数组
			
			Array.prototype.each = function(fn)
			{
				try{
					//1、目的：遍历数组的每一项；
					this.i || (this.i = 0); //给Arr定义一个变量，用于计数：记录当前遍历元素的位置(注：从0开始的) 如果i存在直接用i，如果不存在就初始化i = 0;
					
					//2、判断什以时候去起each方法!
					if(this.length > 0 && fn.constructor == Function) //当Arr数组的长度大于0，并且传进来的fn参数必须是一个函数！
					{
						while(this.i 《 this.length) //循环遍历Arr数组
						{
							var e = this[this.i] ; //把第i(索引位置)个数组的值取出来赋给变量e;
							
							if(e && e.constructor == Array) //判断如果e有值并且是一个数组类型！
							{
								e.each(fn); //直接做递归操作(自己调用自己)
							}
							else
							{
								//如果不是数组(那就是一个单个元素)
								fn.call(e , [e]); //把数组的第e个(当前)元素传递给fn函数，并让函数执行！
							}
							
							this.i++ ;
						}
						this.i = null ; //释放内存，自己写的垃圾回收机制，回收变量！
					}
				}
				catch(ex)
				{
					alert('抛出一个异常(有错误啦)！') ;
				}
				
				return this ; //this 谁调用我，我就属于谁 (现在Arr调用了我，我就属于Arr了)
			}
			
			Arr.each(function(item) //调用each方法循环Arr数组
			{
				alert(item);
			})
