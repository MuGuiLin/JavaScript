JS继承的3种方式

			
			一、原型继承：
						即继承模板，又可以继承原型对象(子类即继承了父类的模板(构造函数)，又继承了父的原型对象)；
						
			二、类继承：
						(只能继承模板，不能继承原型对象【借用构造函数的方式继承】)；
						
			三、混合式继承：
							就是 原型继承 + 类继承(借用构造函数继承)的组合作；【这是常用的继承方法】；
							
							
			//js中怎么去实现继承的：是采用原型链的概念
		
			//在此之前，你应该了解：构造函数、原型对象、实例对象 这3者之间的关系？
			
			/*
			 * 1、构造函数.prototype == 【原型对象】；
			 * 2、原型对象.constructor == 【构造函数的模板】；
			 * 3、原型对象.isPrototypeOf(实例对象)；//判断实例对象的原型，是不是当前对象
			 * 4、构造函数 和 实例对象 == 【类 和 实例】；
			 */
			
			
			
			
			//一、原型继承：即继承模板，又可以继承原型对象(子类即继承了父类的模板(构造函数)，又继承了父的原型对象)
		
			//父类构造函数
			function Preson(name , age , sex)
			{
				this.name = name;
				this.age = age;
				this.sex = sex;
			}
			
			Preson.prototype.ID = 001; //父类的原型对象直接追加属性
			
			//子类构造函数
			function Boy(job)
			{
				this.job = job;
			}
			
			/*
			 * 下面就是原型继承(把父类的实例 赋给 子类的原型对象)
			 * 特点：子类即继承了父类的模板，又继承了父的原型对象！
			 */
			Boy.prototype = new Preson('张三'); //原型继承的方式要在这里传参，这样不好！
			
			var b = new Boy();
			alert(b.name); //通过父类的模板中取得的name值
			alert(b.ID); //通过父类的原型对象取得的ID值
			
			
			
			
			
			//二、类继承(只能继承模板，不能继承原型对象【借用构造函数的方式继承】)
			
			//父类构造函数
			function Mupiao(name , age , sex)
			{
				this.name = name;
				this.age = age;
				this.sex = sex;
			}
			
			Mupiao.prototype.ID = 002;
			
			//子类构造函数
			function Body(name , age , sex , job)//类继承在这里传参，这样虽然解决了上面的问题，【但是它不能访问父类的原型对象】
			{
				//用call apply 方法；
				Mupiao.call(this , name , age , sex ); //用call把Mupiao父类绑定到当前子函数中，并向父类传递对应的参数
				this.job = job;
			}
			
			var B1 = new Body('李四' , 28 , '男' ,'技术总监')
			alert(B1.name);
			alert(B1.age);
			alert(B1.job);
			
			alert(B1.ID);//结果：undefined问题在这里，父类的原型对象并没有被继承！这样也不好！
			
			
			
			
			
			//三、混合式继承：就是 原型继承 + 类继承(借用构造函数继承)的组合作；【这是常用的继承方法】
			
			//父类构造函数
			function Mupiao(name , age , sex)
			{
				this.name = name;
				this.age = age;
				this.sex = sex;
			}
			
			Mupiao.prototype.ID = 003;
			
			//子类构造函数
			function Body(name , age , sex , job)
			{
				Mupiao.call(this , name , age , sex ); //这里就是 借用构造函数继承 的继承方式
				this.job = job;
			}
			
			Body.prototype = new Mupiao(); //这里就是 原型继承 的继承方式
			
			var B1 = new Body('王五' , 19 , '女' ,'技术Boss')
			alert(B1.name);
			alert(B1.age);
			alert(B1.sex);
			alert(B1.job);
			alert(B1.ID);
			
			alert('全部内容：' + B1.name + '; ' + B1.age +'; ' + B1.sex + '; ' + B1.job + '; ' + B1.ID);
