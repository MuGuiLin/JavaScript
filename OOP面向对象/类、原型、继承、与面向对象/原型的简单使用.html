原型的简单使用

			function Preson()
			{
				
			}
			
			//以前的写法
			Preson.prototype.name = '张三'; //给原型加属性
			var p1 = new Preson();
			//alert(p1.name);
			
			
			//现在新的写法
			Preson.prototype = {
				//constructor:Person, //注用这种写法，必须加这句，它是原型对象的构造器
				name:'张三',
				age:24,
				sex:'男',
				job:'程序员',
				say:function()
				{
					alert('我原型的函数！');
				}
			}
			//给原型对象重新设置构造器的方法（要在new 对象之前写）, 它有3个参数，1、重设构造器的对象，2、设置什么属性，3、options配置项
			Object.defineProperty(Preson.prototype , 'constructor',{
				enumerable : false,
				value : Preson
			});
			
			var p2 = new Preson();
			alert(p2.name);
			p2.say();
			alert(Preson.prototype.constructor);
			
			for(m in p2)//循环对象
			{
				alert(m);
			}
			
			
			
			//原型的动态特性
			function mupiao()
			{
				
			}
			
			var m1 = new mupiao(); //简单的原型写法，我可以先(后也可以)new，再赋原型；
			
			mupiao.prototype.say = function()
			{
				alert('我是方法！')
			}
			m1.say();
			
			
			//var m2 = new mupiao(); //下面这种原型的写法，我就不可以先new， 再赋原型；
			mupiao.prototype = { //给原型对象直接再赋一个匿名的对象，
				name : '小穆',
				age : 24,
				alt : function(){
					alert('我是mupiao对象的，原型对象中的函数！');
				}
			}
			var m2 = new mupiao(); //上面这种原型的写法，我要在赋原型之后，再new才行；
			m2.alt();
			alert(m2.name);
			
			//所以，得出 实例对象必须 在 原型对象之后创建! 否则就出错！
