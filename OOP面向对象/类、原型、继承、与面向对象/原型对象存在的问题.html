原型对象存在的问题

			
			//原型的概念：原型对象里的所有属性和方法，被所有的构函 实例化 出来的对象所共享！这也是它的缺点
		
			function mupiao()
			{
				
			}
			
			mupiao.prototype = {
				name : '小穆',
				age : 24 ,
				sex : '男',
				job : '程序员',
				friends :['张三','李四','王五'],
				sayName : function()
				{
					alert('我是原型里面的函数！');
				}
			}
			
			var m1 = new mupiao();
			var m2 = new mupiao();
			
			m1.friends.push('赵六'); //给m1实例追加一个朋友
				
			alert(m1.friends); //结果：多了个赵六
			alert(m2.friends); //结果：同上！【m1追加的属性值不能被单独的提出来！】
			
			//通过上例，看出，虽然只给m1实例追加了一个属性值，但是m2实例也被追加了赵六，这就原型存在的问题，同时也是原型最大的优点！
			
			
			//解决办法：组合使用原型 和 构造函数 式，（这是现在开发的常用方式！）
			function Mupiao (name , age , sex , job , frieds)
			{
				this.name = name; //再次强调：this谁调用我，我就指向谁！
				this.age = age;
				this.sex = sex;
				this.job = job;
				this.frieds = frieds;
			}
			
			//Mupiao构造函数 添加原型
			Mupiao.prototype = {
				constructor : Mupiao,
				sayName : function()
				{
					alert(this.name);
				}
			};
			
			//上面这两种组合 就是常用的开发模式啦！【这样实例出来的属性 和 方法 就不会产生冲突了】
			
			var M1 = new Mupiao('穆飘' , 24 , '男' , '技术总监' , ['小穆','小王','小陈']);
			var M2 = new Mupiao('张敏' , 26 , '女' , '测试员' , ['赵姐' , '李总']);

			/*---------------------常用的两种原型模式-----------------------*/
			
			//一、动态原型模式：（把代码（所有的属性 和 方法）都封装到一起(封装到一个函数内)！）
			function Mupiao (name , age , sex , job , frieds , sayName)
			{
				this.name = name; //再次强调：this谁调用我，我就指向谁！
				this.age = age;
				this.sex = sex;
				this.job = job;
				this.frieds = frieds;
				
				//动态原型方法：
				if( typeof this.sayName != 'function')
				{
					Mupiao.prototype.sayName = function()
					{
						alert(this.name);
					}
				}
			}
			var M1 = new Mupiao('穆' , 24 , '男' , '技术总监' , ['赵','钱','孙' , '李' , '周' , '吴' , '郑' , '王']);
			M1.sayName();
			alert(M1.frieds);
			
			
			
			//二、稳妥构造函数式：durable object（稳妥对象）适用于【非常安全】的环境中!
			//1、没有公共属性，2、不能使用this对象；
			function Preson(name , age , sex)
			{
				var Obj = new Object(); // 创建一个要返回的对象
				
				var sex = '女'; //定义私有变量，注：外面不能访问这个变量，可通过下面的函数访问
				var age = age;
				var job = 'UI设计师';
				
				Obj.sayName = function()
				{
					alert(name + sex + age + job);
				}
				
				return Obj;
			}
			var P1 = Preson('小星星',26);
			P1.sayName();
			alert(P1.name); //找不到 undefined
