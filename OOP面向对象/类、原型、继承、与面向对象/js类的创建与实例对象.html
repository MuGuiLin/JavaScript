类的创建于实例对象

			面向对象的语言有一个标志，那就是他们都有类的概念，
			通过类可以创建任意多个具有相同属性和方法的对象。
			但是在ECMAScript中没有类的概念，但是我们可以通过其他方式来模拟面向对象的类。
			工厂模式：工厂模式是软件工厂领域中一种广为人知的设计模式。
			构造函数模式：比如像ECMAScript中的Array、Object、Date等都是通过构造函数来创建的。
			
			
			
			//普通对象
			var obj = new Object();
			obj.name = '张三';
			obj.age = 24;
			obj.sex = '男';
			obj.sayName = function()
			{
				alert('我是张三！');
			}
			
			//类的概念 先通过以下两种方式来模拟js面向对象中的类
			
			//1、简单的工厂模型;
			function createPerson(name,age,sex)//把函数变为类
			{
				var obj = new Object();
				obj.name = name;
				obj.age = age;
				obj.sex = sex;
				obj.sayName = function()
				{
					alert(name);
				}
				return obj;
			}
			
			var p1 = new createPerson('小穆',25,'男');//通过createPerson()实例化出一个人
			var p2 = new createPerson('小菜',24,'女');//通过createPerson()实例化出两个人
			alert(p1.name);
			
			
			//2、构造函数式 ；【js特有的】
			function Preson(name,age,sex)//方法名首字母大写：表示类的模板，里面不用new 直接用this
			{
				this.name = name;
				this.age = age;
				this.sex = sex;
				this.sayName = function()
				{
					alert(this.name);
				}
			}
			
			//构造一个对象，new关键字，传递参数，执行模板（类）代码，返回对象;
			var p1 = new Preson('小锁',23,'女');
			var p2 = new Preson('小张',24,'女');
			
			alert(p1.age);
			p1.sayName();
			
			alert(p1 == p2);//结果为：false，由此可见这就是类的概念：根据同一个模板创建出不同的实例对象!
		
			alert(p1 instanceof Preson); //判断 p1 是不是 Preson 的实例；
			
			
			
			//3种创建类的实例的方法：
			
			//1、当作构造函数去使用
			var p1 = new Preson('小锁',23,'女');
			
			//2、作为普通的函数去调用
			Preson('小锁',23,'女');//此时preson的执行环境就是wineow.Preson了,这仅仅是普通函数的用法
			
			//3、在另一个对象的作用域中调用,利用 call 或 apply 方法
			var o = new Object();
			Preson.call(o,'穆飘',25,'男'); //通过call(函数方法) 把Preson对象绑定到o对象，此是o对象就有了Preson对的属性和方法啦!
			alert(o.name);
