			//在此 先创建一个方法：用于 实现只继承1次父类模板 目的：实现只继承父类的原型对象，不继承父类的模板！
			function HeirMethod(Subclass, Fatherclass)//传两个参数：1、子类的构造函数，2、父类的构造函数
			{
				var Fn = function(){}; //1、创建一个空函数，用于中转父的原型对象
				Fn.prototype = Fatherclass.prototype // 2、把父类的原型对象 赋给 上面空函数Fn的原型对象 【此时就屏蔽掉了父类的模板啦！】
				Subclass.prototype = new Fn(); //3、再把父类的实例 赋给 子类的原型对象！【在此基本上就达到目的啦！】
				
				Subclass.prototype.constructor = Subclass // 还原子类的构造器；
				
				//保存一个父类的原型对象：用于 一方面方便解耦；另一方面获得父类的原型对象
				Subclass.SubAttr = Fatherclass.prototype; //给子类定义一个静态属性 用于 接收父类的原型对象
				
				//判断父类的原型对象的构造器(加保险)
				if(Fatherclass.prototype.constructor == Object.prototype.constructor)
				{
					Fatherclass.prototype.constructor = Fatherclass; //还原父类原型对象的构造器
				}
			}
			
			//父类模板（构造函数）
			function Person(name , age , sex , job)
			{
				this.name = name;
				this.age = age;
				this.sex = sex;
				this.job = job;
			}
			
			//父类的原型对象
			Person.prototype = {
				constructor : Person, //把构造器指向Person
				tizhong : '68kg',
				height : '172cm',
				aihao : function()
				{
					alert('喜欢打篮球！');
				}
			}
			
			//子类模板（构造函数）
			function Boys(name , age , sex , job, friends , sheep , skill)
			{
				//Person.call(this , name , age , sex , job);//绑定父类模板对象到当前子类中来【现在不绑定Preson了】
				Boys.SubAttr.constructor.call(this , name , age , sex , job);//【绑定上面 HeirMethod 方法中的 自定义的子类属性 SubAttr】这就是方便解耦
				this.friends = friends;
				this.belong = sheep;
				this.skill = function()
				{
					alert('我会' + skill);
					return skill;
				}
			}
			
			//子类原型对象
			Boys.prototype = {
				aihao : function()
				{
					alert('喜欢上网！');
				}
			}
			
			//Boys.prototype = new Person();//继承父类模板【现在不用这种继承方式了】
			
			HeirMethod(Boys , Person); //【用这个方法：来代替上面这种继承方式，它需要传2个参数：1、子类的构造函数，2、父类的构造函数】
