JS原型常用方法

			1、实例对象名.constructor //返回对创建此对象的数组函数的引用(保存对象中的创建函数)。
			
			2、object1.isPrototypeOf(object2));  //判断检测传入的对象是否是另外一个对象的原型：true:是，false:否（判断指定对象object1是否存在于另一个对象object2的原型链中，是则返回true，否则返回false）
			
			3、Object.getPrototypeOf(实例对象名); //根据实例对象获得对应的原型对象
			
			4、实例对象名.hasOwnProperty('属性名') //用于查找判断对象中是否有这个属性：true:有，false:没有
			
			5、'属性名' in 实例对象名 //in操作符 判断一个对象的属性，是否在实例对象或原型对象中
			
			6、Objent.getOwnPropertyNames(对象名) //枚举对象的所有属性，不管该内部属性能否(注：constructor属性是不能被枚举的)被枚举！
			
			
			//Object.getPrototypeOf(); 根据实例对象获得原型对象
		
			function Preson()
			{
				
			}
			
			Preson.prototype.name = '小穆'; //通过原型对象给Preson对象设置属性
			Preson.prototype.age = 20;
			Preson.prototype.sex = '男'
			Preson.prototype.sayName = function()
			{
				alert('我是原型对象的方法');
			};
			
			var p1 = new Preson();
			alert(p1.name);
			
			var prototypeObj = Object.getPrototypeOf(p1);  // 根据实例对象 p1 , 获得它(p1)的原型对象
			
			if(prototypeObj === Preson.prototype)
			{
				alert('p1的原型对象是：Preson.prototype')
			}
			alert(prototypeObj);
			
			
			/*
			 * 注意下面的代码
			 * 当代实例化一个对象后，它会先找当前实例代的属性，然后再去找它的原型对象；
			 * 如果当前实例对象中的属性和原型对象中的属性相同，那到原型对象中的属性会被前实例对象中的属性覆盖掉；
			 * 如果当前实例对象中没有属性，那么再到原型对象中去找属性；
			 * 如果都没有就返回undefined;
			 */
			var p2 = new Preson();
			p2.name = '张三';
			alert(p2.name); //覆盖掉
			alert(p2.age); //原型对象的属性
			alert(p2.mupiao); //都没有
			
			
			/*
			 * 小技巧：
			 * 但是我又只想要原型对象中的属性怎么办？
			 */
			var p2 = new Preson();
			p2.name = '张三';
			delete p2.name; //先删除实例对象中的属性，
			alert(p2.name); //此时：就不会被覆盖掉，弹出的就是 原型对象中的属性了！
	
	
			/*
			 * 如何判断一个对象的属性，是否属于实例对象的属性  //p3.hasOwnProperty('name')
			 * 扩展：如何判断一个对象的属性, 是属于实例对象的属性，还是原型对象的属性
			 */
			var p3 = new Preson();
			alert(p3.name);//p3.name 是来自于原型对象中的属性
			//p3.name = '123';
			alert(p3.hasOwnProperty('name'));//hasOwnProperty('属性名') //用于查找判断对象中是否有这个属性：true:有，false:没有
			
			
			/*
			 * 如何判断一个对象的属性，是否在实例对象或原型对象中 'name' in p4  in操作符
			 */
			var p4 = new Preson();
			//p4.age = 25;
			alert('name' in p4);// 有：true
			alert('aaa' in p4); // 没有：false
			
			
			/*
			 * 扩展：自定义的方法
			 * 如何判断一个属性是否属于原型对象 并且 是否在原型对象中
			 */
			function hasPrototypePrototypeYN (Object , name)
			{
				return !Object.hasOwnProperty(name) && name in Object ; //这是关键点 同时用到了hasOwnProperty() 和 in
			}
			alert('答案是：' + hasPrototypePrototypeYN(p4,'age'));
			
			
			/*
			 * ECMA5中的新特性 Objent.keys();
			 * 用于获取当前对象(实例对象或原型对象)中所有的ksys(属性名) 返回的是一个数组的形式；
			 */
			var p5 = new Preson();
			p5.name = '穆飘';
			p5.age = 24;
			p5.sex = '男';
			var attributes = Object.keys(p5); //取得实例对象中所有的keys(属性)
			alert(attributes);
			
			var attributes2 = Object.keys(Preson.prototype);
			alert(attributes2);
			
			
			/*
			 * Objent.getOwnPropertyNames() //和上面差不多，枚举对象的所有属性，不管理该内部属性能否(注：constructor属性是不能被枚举的)被枚举！
			 */
			var mupiao = Objent.getOwnPropertyNames(Preson.prototype)
			alert(mupiao);
