JSON与XML相互转换

			主要是借用 以下:jquery 的这3个文件！ 下载地址
				1、jquery.js
				2、jquery.json2xml.js
				3、jquery.xml2json.js
			
			一、先下载并引入以下3个文件
			
			
			
			
			二、XML字符串 转换 成JSON对象
				
				$.xml2json()方法；
				
			三、JSON 对象 转换 成XML字符串
			
				$.json2xml()方法；
				
				
			实例：
				
				/*
				 * 一、XML字符串 转换 成JSON对象
				 */
					
					//XML字符串
					var str = 	""+
									"小穆"+
					 				"25"+
					 				"小明"+
					 				"小强"+
					 				"小正"+
					 			"";
					console.log(str);
					alert(str);
					
					
					var obj = $.xml2json(str);//将符合XML语法规范的字符串 转换为 json对象；
					console.log(obj);	
					alert(obj);
					
					
					var jsonstr = JSON.stringify(obj);//将json对象 转换 为字符串；
					console.log(jsonstr);
					alert(jsonstr);
					
				
				
				/*
				 * 一、JSON 对象 转换 成XML字符串
				 */
				
					var Obj = {
						name : "穆飘",
						age : 24,
						sex : "男",
						friends :["小王" , "小明" , "小增"]
					};
					
					var xmlstr = $.json2xml(Obj);
					console.log(xmlstr);
					alert(xmlstr);
