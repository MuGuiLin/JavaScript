JS函数的参数

			JS函数的参数对象：arguments.length实参个数，arguments.callee.length形参个数；
		
			/function 参数 分为：【实参，和  形参（可以任意思设置）】
					  //形参
			function myfn(a,b,c,d,e)
			{
				//alert(myfn.length); //形参的个数(不推荐)
				alert(arguments.callee.length)//形参的个数(推荐)
				
				alert(arguments.length);//实参的个数 用arguments对象【注它只能在函数内部使用】
				return a + b;
			}
					  //实参
			alert(myfn(10,10));
			
			//在js中实参 与 形参 这间的个数可以不相等，也不会报错！不过最好相等为妙!
			
			//arguments.length用于判断实参个数的长度，用得最多的是做递归操作(arguments.callee  意思是函数它本身)！
			
			//在高级语言下形参与实参必须相等，但在js中可以不用相等，如果要严格举例如下：
			
			function myFn(a,b,c,d,e)
			{
				if(arguments.length == arguments.callee.length)
				{
					alert('您好：此函数的形参 和 实参个数是【相等】的！');
				}
				else
				{
					alert('对不起：此函数的形参 和 实参个数【不相等】的！');
				}
			}
			myFn(1,2,4,5,7)
			
			
			function myFN(a,b,c,d,e)
			{
				if(arguments.length == arguments.callee.length)
				{
					return a + b;
				}
				else
				{
					return '参数不正确!';
				}
			}
			alert(myFN(1,2,4,5));
