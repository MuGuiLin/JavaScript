函数的链式调用

			javascript对方法进行链式调用的能力。如果对jquery有一定了解，那么很明显你一定知道这是一个什么意义。
			
			简单函数链式调用。 return this；
			

			
			//模拟jquery底层链式编程
		
		
			/*
			 * 这是一个块级作用域:(闭包)
			 * 特点：
			 * 		1、立即执行（当刚打开程序的时候，里面的代码直接被执行）
			 * 		2、内部成员变量，外部无法访问（除了前面不加 var 的变量【全局变量】）
			 * 
			 * 		alert('我被立即执行了！');
			 * 		a = '我是全局变量，我没加var';
			 */
			(function(window , undefined)
			{
				// $ 符号最常用的对象，返回给外界。
				// 在大型程序中做开发时，一般使用'_'下划线作为【私有的对象】（这是规范，你要慢慢养成）
				
				//创建一个对象 _$ 【外部不能访问】
				function _$(arguments)
				{
					
				}
				
				//在Function类上扩展一个可以实现链式编程的方法
				Function.prototype.mupiao = function(methodName , fn)
				{
					this.prototype[methodName] = fn;
					return this; //链式编程必须加
				}
				
				//给_$加一些公共的属性、方法
				_$.prototype = {
					
					constructor : _$, //还原构造器
					_URL : 'http://www.baidu.com',
					addEvent : function()
					{
						alert('我是事件方法！');
						return this;
					},
					setStyle : function()
					{
						alert('我是样式方法！');
						return this;
					},
				}
				
				//把_$ 再赋给全局变量 window.$ 【而现在我在外部调用&时就相当于调用_$了，并且有了_$所有的属性和方法】
				window.$ = _$;
				
				
				//写一个准备方法
				_$.Ready = function(fn)
				{
					window.$ = function()
					{
						return new _$(arguments);
					}
					
					fn();
					
					//链式编程
					_$.mupiao('addEvent' , function(){
						
					}).mupiao('setStyle' , function(){
						
					});
				}
			})(window); //程序入口, window传入作用域中
			
			
			
			//外部调用
			$.Ready(function()
			{
				alert($());
				$('#inp').addEvent().setStyle();
			})
