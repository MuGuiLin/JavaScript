函数的链式调用

			javascript对方法进行链式调用的能力。如果对jquery有一定了解，那么很明显你一定知道这是一个什么意义。
			
			简单函数链式调用。 return this；
			
			
			//简单的函数链式调用(链式编程)
			
			
			
			function Dog(){
				
				this.run = function(){
					alert('我是跑的方法！函数链式调用方式');
					return this; //返回当前对象
				}
				
				this.eat = function(){
					alert('我是吃饭的方法！函数链式调用方式');
					return this;
				}
				
				this.sleep = function(){
					alert('我是睡觉的方法！函数链式调用方式');
					return this;
				}
			}
			
			var D1 = new Dog();
			D1.run().eat().sleep(); //一次性调用3个方法
