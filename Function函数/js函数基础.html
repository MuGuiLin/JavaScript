函数基础

			//在js中，函数是由事件驱动的或者当它被调用时执行的可重复使用的代码块，它也是一种数据类型。
		
			
			
			//匿名函数
			document.getElementById('but').onclick = function()
			{
				window.alert("我是匿名我没函数名！。。。。");
			}
			
			//简单函数：function关键字+函数名(没有返回值)
			function test1()
			{
				alert("我是通过函数弹出来的！");
			}
			test1();
			
			//立即执行函数
			(function mupiao()
			{
				alert('我不需要调用，我可以立即执行');
			})();
		
			//带参数的函数：function关键字+函数名(参数1(返回值)，参数2(返回值)。。。。。)
			function test2(csa,csb)
			{
				//alert(csa + csb);
				return csa + csb;
			}
			var sum = test2(12,10);
			alert(sum);
			
			//函数也是一种数据类型，(function类型),并且很灵活！如下实例：
			function myfn1(fn)
			{
				alert('我是函数myfn1 ')
				fn();
			}
			function myfn2()
			{
				alert('我是函数myfn2的被执行了！');
			}
			
			myfn1(myfn2);//函数可以作为参数传，可以在函数a中调用函数b;
			
			myfn1(function()
			{
				alert('在函数中还可以再放匿名函数！');
			});
			
			//在函数中还可以再嵌套函数
			function myfn3()
			{
				alert('我是函数myfn3');
				
				function myfn4()//注：myfn4只能在myfn3里面调用，外面调不到！
				{
					alert('我是函数myfn4！。。。');
				}
				
				myfn4();
			}
			myfn3();
