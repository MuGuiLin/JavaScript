函数的3种定义方式

			//在js中，函数是由事件驱动的或者当它被调用时执行的可重复使用的代码块，它也是一种数据类型。
		
			//函数的3种定义方式
			
			//1、function 语句式  【js解析器会优先执行存入内存中，所以可以先调用，再执行函数体】
			
			FnName1()//这是调用方法，可以放到前面；
			function FnName1()
			{
				alert('我是第1种定义方式【function语句式！】');
			}
			
			//2、函数的直接量 【ECMA script推荐用这种】【js解析器按照顺序执行,要先创建，再调用】
			var FnName2 = function()
			{
				alert('我是第2种定义方式【函数直接量】！')
			}
			FnName2();//这是调用方法 ，只能放在后面；
			
			//3、function 构造函数式 【js解析器按照顺序执行,要先创建，再调用】
			var FnName3 = new Function('a','b','return a+b;'); 
			
			alert(FnName3(10,12));//这是调用方法，只能放在后面；
			
			
			//执行效率的比较
			var d1 = new Date();  //日期对象
			var t1 = d1.getTime();  //获取函数创建之前毫秒数
	
			for(var i = 0; i 《= 10000; i++)
			{
				//function Fn1(){;};//函数语句式【3毫秒；因为它只被创建1次，然后存放在内存中，占用内存】执行效率快，它是静态的！。。。
				//var Fn2 = function(){;};//函数语句式【3毫秒；因为它只被创建1次，然后存放在内存中，占用内存】执行效率快，它是静态的！。。。
				var Fn3 = new Function(); //直接变量式的空函数【36毫秒左右；它执行次就 new 1次后消毁了，不占用内存】执行效率慢，它是动态的！。。。
				
				//所以上效率来看，function语句式 比 直接变量式快
			}
			
			var d2 = new Date();  //日期对象
			var t2 = d2.getTime();  //获取函数创建之后毫秒数
		
			alert(t2-t1 );//最后比效以上两种函数定义方式所用的时间！【多刷新几次就看到了！】
			
			
			//函数的解析顺序
			function f(){return 1;} 				// 函数1 	
			alert(f());		//返回值为4 说明第1个函数被第4个函数覆盖	
			
			var f = new Function("return 2;");		// 函数2 
			alert(f());		//返回值为2 说明第4个函数被第2个函数覆盖
			
			var f = function(){return 3;}			// 函数3 
			alert(f());	       //返回值为3 说明第2个函数被第3个函数覆盖
			
			function f(){return 4;} 				// 函数4 
			alert(f());		//返回值为3 说明第4个函数被第3个函数覆盖
			
			var f = new Function("return 5"); 		// 函数5 
			alert(f());	//返回值为5 说明第3个函数被第5个函数覆盖
			
			var f = function(){return 6 ;}			// 函数6 
			alert(f());		//返回值为6 说明第5个函数被第6个函数覆盖
	
			
			//函数的作用域
			var k = 1 ; 
			function t1()
			{
				var k = 2 ; 
				function test(){return k ;} //function语句式， 结果：2  【函数(局部)作用域】
				var test = function(){ return k}; //函数直接量式   结果：2 【函数(局部)作用域】
				var test = new Function('return k;');//构造函数式   结果：1 【顶级作用域】//它相当于在函数外创建的，所以它不认识函数里面的k;
				alert(test());
			}
			t1();
			
			//通过以上3种不同的函数创建方式比较，构造函数式是顶级作用域，虽然在私有函数中创建，但它也只认识外面的变量k
