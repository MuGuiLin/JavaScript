JS接口

			接口是面向对象javascript程序员的工具箱中最有用的工具之一，
			javascript没有定义接口的概念，我们通过模拟高级程序语言的方式来创建javascript中的接口。
			接口的概念：提供一种以说明一个对象应该有哪些方法的手段。

			//三、鸭式辨型的方式：【强烈建议使用】
			//核心：一个类 实现接口的主要目的：就是把接口里面的方法都实现（检测方法）;
			//优点：完全面向对象，代码也实现统一、整合了，也解耦了！
			
			//1、自定义一个接口类，用于实例化N多个接口：
			var Interface = function(name , methods)//它需要传2个参数：1、接口的名字name(字符串)，2、接收方法名methods的集合(数组)
			{
				if(arguments.length != 2)
				{
					alert('实例的接品构造器里面的参数长度必须是2个');
					throw new Error('实例的接口构造器里面的参数长度必须是2个！');
				}
				this.name = name;
				this.methods = [];
				for(var i = 0; i 《 methods.length; i++)
				{
					if(typeof methods[i] !== 'string')
					{
						throw new Error('方法里面的名字必须为字符串类型！');
					}
					
					this.methods.push(methods[i]);
				}
			}
			
			//2、准备工作：
			
			//①、实例化接口对象
			var BasicInterface = new Interface('BasicInterface' , ['insters' , 'updates' , 'deletes']);
			var AlertInterface = new Interface('AlertInterface' , ['Alert','hello']);
			
			//②、实现具体的类
			var BasicClass = function()
			{
			};
			
			//③、实现接口的方法
			BasicClass.prototype.insters = function(obj){
				alert('新增方法！');
			}
	//		BasicClass.prototype.updates = function(obj){
	//			alert('修改方法！');
	//		}
	//		BasicClass.prototype.deletes = function(obj){
	//			alert('删除方法！');
	//		}
			
			var AlertClass = function()
			{	
			};
			
			AlertClass.prototype.Alert = function(obj){
				alert('obj');
			}
			AlertClass.prototype.hello = function(obj){
				alert('你好啊！');
			}
			
			//3、检验接口里面的方法是否实现：
			Interface.TestInterface = function(object)
			{
				if(arguments.length 《 2) //传进来的实参至要两个参数
				{
					alert('接收的参数不能小于2个！');
				}
				
				for(var i = 1; i 《 arguments.length; i++) //var i = 1;因为第1参数是类，这里要的接口
				{
					var interfaceObj = arguments[i];
					if(interfaceObj.constructor !== Interface)
					{
						alert('传递的参数，不是接口类的类型！');
					}
					for(var j = 0; j 《 interfaceObj.methods.length; j++)
					{
						var methodName = interfaceObj.methods[j];
						if(!object[methodName] || typeof object[methodName] !== 'function')
						{
							alert('没有找到 ' + methodName + ' 方法，请仔细验证您的代码！' );
						}
					}
				}
			}
			
			var B1 = new BasicClass();//实例化类
			Interface.TestInterface(B1 , BasicInterface);//检测 类 和接口
			B1.insters();
			
			var B2 = new AlertClass()
			Interface.TestInterface(B2 , AlertInterface)
			B2.Alert();
			B2.hello();
