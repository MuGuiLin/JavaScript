Object 基础

			Object对象是所有数据类型的基础类（根类）;
			
			//创建(实例化)对象的三种方法：
				①、var obj = new Object();
				
				②、var obj = {
					name : '小穆',
					age : 24,
					sex : '男',
					fnName : function()
					{
						alert('我是方法@！');
					}
				}
				alert(obj.fnName());
				
				③、var obj = {};
				
				//给对象设置属性
				obj.name = '张三';
				obj.age = 24;
				obj.weight = '65kg';
				//或
				obj['height'] = '168cm';
				obj['sex'] = '男';
		
				
				
				obj.name = '穆飘';//修改属性
				
				//给对象设置方法
				obj.mupiao = function()
				{
					alert("我的职业：Web开发！");
				}
				
				obj.QQ = function()
				{
					alert('我的QQ：914026091');
				}
				
				//访问对象的属性或方法
				alert(obj.name);
				alert(obj.age);
				obj.mupiao();
				//或如下
				alert(obj.name + obj.age + obj.mupiao());
				
				//delete 操作符：用来删除对象的属性或方法的！
				delete obj.name; //删除属性
				delete obj.mupiao;//删除方法(注：删除方法时，后面不要加()括号)
				alert(obj.name);
				obj.mupiao();
				
				以上是对对象的增、删、查、改、的操作方法！
				
				
				
				//如何去循环对象，用 for in 语句
				for(var m in obj)
				{
					alert(m +' : '+ obj[m])
					alert(obj[m]);//输出对象的属性和方法
				}
