Object 常用方法

			var obj = {};
			obj.name = "小穆";
			obj['sex'] = '男';
			obj.age = 24;
			obj.it = function()
			{
				alert('我是做IT的！');
			}
			
			对象名.constructor;//constructor保存对象中的创建函数
			例：
				var mupiao = obj.constructor;
				alert(obj.constructor);
				alert(mupiao);
				
			对象名.hasOwnProperty('属性名');//用于查找对象中是否有这个属性：true:有，false:没有
			例：
				obj.hasOwnProperty('sex');
			
			对象名.propertyIsEnumerable('属性名'); //用于检测传对象指定属性能否被 for in 所枚举(循环)出来：true:可以，false:不行
			例：
				obj.propertyIsEnumerable('name');
				alert(obj.propertyIsEnumerable('name'));
			
			对象名.isPrototypeOf();//用于检测传入的对象是否是另外一个对象的原型：true:是，false:否
			例：
				alert(obj.isPrototypeOf());
				
			obj.toLocaleString());//返回对象的字符串表示，该字符串与执环境的地区对应
			
			obj.toSource());//返回对象的字符串表示
			
			obj.valueOf());//返回对象的字符串、数值、布尔表示
