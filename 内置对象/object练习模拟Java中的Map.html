				function Map()
				{
					//private
					var obj = {};//创建一个空对象
					
					//put方法 用于把键值对绑定到obj对象上
					this.put = function(key,value)
					{
						obj[key] = value; 
					}
					
					//eachMap方法 用于循环Map
					this.eachMap = function(fn)//传一个回调函数
					{
						for(var a in obj)
						{
							fn( a, obj[a]);
						}
					}
					
					//size方法 用于查看Map的个数
					this.size = function()
					{
						var mupiao = "";
						for(var m in obj)
						{
							mupiao++;
						}
						return 'Map共有：' + mupiao + "个键值";
					}
					
					//get方法 用于根据key 取得对应的 value
					this.get = function(key)
					{
						if(obj[key] || obj[key] === 0 || obj[key] === false)//如果键中的值是0 或者是 flase 也返回key
						{
							return key +' 对应的值：'+ obj[key];
						}
						else
						{
							return null;
						}
					}
					
					//remove方法 用于根据key 删除对应的 value
					this.remove = function(key)
					{
						if(obj[key] || obj[key] === 0 || obj[key] === false)
						{
							alert(obj[key] + '：删除成功了!');
							delete obj[key];
							
						}
						else
						{
							return false;
						}
					}	
				};
				
				//模拟Java中的Map
				var m = new Map();
				m.put('01','abc');
				m.put('02',123);
				m.put('03',true);
				m.put('04',new Date());
				m.put('05',new Date());
				
				//调用Map中的方法
				alert(m.size());//查看Map中的个数
				alert(m.get('02'))//指定查看Map中的值
				m.remove('04')//指定删除Map中的值
				alert(m.get('04'));//指定查看Map中的值

					alert(key +':'+ value);
