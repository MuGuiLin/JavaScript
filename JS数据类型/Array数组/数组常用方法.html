JS数组常用方法

			什么JS是数组：JS数组就是一组数据的组合（集合），它是一个对象，长度可随意改变，可同时存放任意类型，它的下标从0开始计算！
			
			组的定义方式：2种
			①、var arr1 = new Array([1,2,3,4,true,'abc',new,undefined]);//使用new关键字。
			②、var arr2 = [1,2,3,4,true,'abc',new,undefined];//直接一个[]中括号，推荐使用这种方式。
			
			var arr [1,2,3,4,true,'abc',new Date(),undefined]; //一维数组；
			var arr [1,2,3,['abc',45,123,new Date()],4,true,'abc',new Date(),undefined]; //二维数组；
			var arr [1,2,3,4,true,'abc',new Date(),['abc',45,[123,'abc',true,'hello'],123,new Date()],undefined];  //三维数组；
			
			数组常的方法：length，push()，pop()，shift()，unshift()，splice()，slice()，concat()，join()，sort()，reverse()；
			
			1、数组名.length; //该方法计算数组中的长度（就是计算有多少个数据！）。
			例：
				alert(arr2.length = 5);
			
			2、数组名.push(); //该方法向组中追加（推送）数据。
			例1：
				var Arr = [];//定义了一个空数组
				Arr.push(1,2,3,4,5,true,'a','b','c','abc'); //追加（推送）数据
				alert(Arr);
			例2：
				//向数组中添加数据(并返回组数的新长度)
				var Arr = ['你好',12];
				var nArr = Arr.push(1,2,3,4,5,true,'a','b','c','abc');
				alert(nArr);
				
			3、数组名.pop(); //该方法从数组的尾部移除一个元素
			例：
				var Arr = [1,2,3,4,5,true,'a','b','c','abc'];
				var shanchu = Arr.pop();//删除，并返回被删除的元素
				alert(Arr);
				alert(shanchu + '被删除了');
				
			3、数组名.shift();//从数组的头部移除一个元素
			例：
				var Arr = [12,51,253,4.5,665,true,'a','b','c','abc'];
				var shanchuTB = Arr.shift();//删除，并返回被删除的元素
				alert(Arr);
				alert(shanchuTB + '被删除了');
				
			4、数组名.unshift();//从数组的头部插入一个或多个元素
			例：
				var Arr = [12,51,253,4.5,665,true,'a','b','c','abc'];
				var tbadd = Arr.unshift('你好','我是新添加','的哈哈');//删除，并返回被删除的元素
				alert(Arr);
				alert(tbadd); //打印组数的新长度
				
			5、数组名.splice(参数1,参数2,参数3); //去除、替换 数组中的元素
			例1：	
				var Arr = [1,2,3,4,5,6];
				//Arr.splice(1,2,3,4,5,6);//第1个参数：去除起始位置，第2个参数：表示去除元素的个数，第3个参数之后：表示追加新元素的个数
				//alert(Arr);
			例2：	
				//去除数组中的元素（功能的扩展,替换）
				var Arr = ['mupiao',828,false,'哈哈',123,'Hello',true,'JS'];
				//Arr.splice(0);//全部去除完了
				//Arr.splice(1)//除了第1元素以外，其他都去除
				//Arr.splice(2,1);//把false去除了
				Arr.splice(0,3,'MUPIAO',282,true);//把'mupiao',828,false,替换为 'MUPIAO',282,true
				alert(Arr);
				
			6、数组名.slice(参数1,参数2);//截取数组中的元素【不操作数组本身】
			例：
				var Arr = ['mupiao',828,false,'哈哈',123,'Hello',true,'JS'];
				var jiequ = Arr.slice(3,5);//把'哈哈',123,提取出来，参数1,参数2：是左闭右开区间的，注：下标是从0开始的！
				alert(jiequ);
				
			7、数组名1.concat(数组名2);//合并数组【不操作数组本身】
			例：
				var Arr1 = [1,2,3,4,5,6];
				var Arr2 = ['a','b','c','d','e','f'];
				var hebing = Arr1.concat(Arr2);//合并数组
				var NewArr = [hebing];//存入到新数组
				alert(NewArr);
				
			8、数组名.join('要加入的内容');//在数组每个元素之间加入内容【不操作数组本身】
			例：	
				var Arr = [1,2,3,4,5,6,7,8];
				var jiaru = Arr.join('——')
				alert(jiaru);
				
			9、数组名.sort();//常正排序(注：排序是先匹配第1个字符);
			例：
				var Arr = [8,4,6,1,5,3] //常正排序
				var Arr = ['a',"d",'r','q','w']//常正排序
				var Arr = [8,4,6,25,5,21] //排序是先匹配第1个字符
				Arr.sort();
				alert(Arr);
				
			10、数组名.reverse();//位置倒排【颠倒 最后变到最前】（注：是数组元素的【位置】倒排）
			例：
				var Arr = [8,4,6,1,5,3] //位置倒排
				var Arr = ['a',"d",'r','q','w']//位置倒排
				var Arr = [8,4,6,25,5,21] //位置倒排
				Arr.reverse();
				alert(Arr);
