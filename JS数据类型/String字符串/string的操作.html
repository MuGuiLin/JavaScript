JS string的操作

			var str = "m|u|p|iao2008abc.jpg";
		
			alert(str.length); // 输出字符串的长度
			
			alert(str.substr(str.length - 3)); //substr(index)  截取字符串的最后3个字符
			
			alert(str.substring(6 , 10)); //substring(index , index) 截取指定的字符 它是最常用到的字符串截取方法，它可以接收两个参数(参数不能为负值)
			
			alert(str.slice(-1 , 2)); //substring(index , index) 则该参数规定的是从字符串的 【尾部开始算】起的位置，它可以接收两个参数(参数可以为负值)
			
			alert(str.charAt(2)); //charAt(index) 返回指定位置的字符
			
			alert(str.charCodeAt(2)); //charCodeAt(index)返回指定位置的字符的Unicode编码
			
			//alert(str.fromCharCode(97, 98, 99, 100, 101)); //charCodeAt(index)返回指定位置的字符的Unicode编码
			
			alert(str.indexOf('a'));//indexOf() 返回某个指定的字符串值在字符串中首次出现的位置 。它可以接收两个参数，第1个表示要查找的子字符串，第2个表示查找的开始位置，省略的话则从开始位置进行检索。方法对大小写敏感,如果没检索到，则返回 -1。
			
			alert(str.lastIndexOf('a'));//indexOf() 返回某个指定的字符串值在字符串中最后出现的位置 。它可以接收两个参数，第1个表示要查找的子字符串，第2个表示查找的开始位置，省略的话则从开始位置进行检索。方法对大小写敏感,如果没检索到，则返回 -1。
			
			alert(str.replace('2008abc' , '2018ABC')); //用来进行字符串替换操作，它可以接收两个参数，前者为被替换的子字符串（可以是正则），后者为用来替换的文本
			
			alert(str.split('|')); //split() 方法用于把一个字符串分割成字符串数组。第一个参数表示分割位置(参考符)，第二个参数表示返回数组的允许最大长度(一般情况下不设置)。
			alert(str.split(''));
			
			alert(str.toLowerCase()); //toLowerCase() 把字符串中的大写字母转换为小写，
			
			alert(str.toUpperCase());//toUpperCase() 把字符串中的小写字母转换为大写。
